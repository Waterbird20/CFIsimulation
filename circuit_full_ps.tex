\newlength{\sqsize}

\newlength{\mqheight}

% Zero-declared-size wrapper so gate height is driven by minimum height, not content
\newcommand{\mgl}[1]{\makebox[0pt][c]{\rotatebox{90}{$\scriptstyle #1$}}}

\newcommand{\gl}[2]{%
  \makebox[\sqsize][c]{%
    \vbox to \sqsize{\vfil
      \baselineskip=0pt\lineskip=1pt\lineskiplimit=0pt
      \hbox to \sqsize{\hfil$\tiny #1$\hfil}%
      \hbox to \sqsize{\hfil{\fontsize{5}{6}\selectfont$#2$}\hfil}%
    \vfil}%
  }%
}

\tikzset{
  operator/.append style={rounded corners=0pt, inner sep=0pt},
  sq-gate/.style={minimum size=\sqsize, inner sep=0pt, text=white, draw=none},
  mq-gate/.style={minimum width=\sqsize, minimum height=\mqheight, inner sep=0pt, text=white, draw=none},
  % Redefine meter to match Qiskit style: gray fill, black arc only
  meter/.style={
    draw=none, fill=qgray, rectangle, rounded corners=0pt,
    minimum size=\sqsize, inner sep=0pt,
    font=\vphantom{A},
    path picture={
      \clip (path picture bounding box.south west)
            rectangle (path picture bounding box.north east);
      % Semicircle arc (∩ shape), radius 0.2 centered at (0, -0.065)
      \draw[black, line width=1.2pt]
            ([shift={(-0.2, -0.065)}]path picture bounding box.center)
            arc (180:0:0.2);
      % Needle: straight line from arc center toward upper-right
      \draw[black, line width=1.2pt]
            ([shift={(0, -0.065)}]path picture bounding box.center)
            -- ([shift={(0.2, 0.1414)}]path picture bounding box.center);
    }
  }
}
